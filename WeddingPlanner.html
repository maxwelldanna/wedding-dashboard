    <script>

        let venues = [

            { id: 1, name: "The Grand Ballroom", capacity: 250, rental: 6000, catering: 155, vibe: 9, log: 8 },

            { id: 2, name: "Riverside Garden", capacity: 150, rental: 4500, catering: 95, vibe: 7, log: 5 },

            { id: 3, name: "Modern Loft Space", capacity: 100, rental: 7000, catering: 130, vibe: 8, log: 9 }

        ];


        let weights = { cost: 0.4, vibe: 0.4, log: 0.2 };


        function updateWeights() {

            const c = parseInt(document.getElementById('w-cost').value);

            const v = parseInt(document.getElementById('w-vibe').value);

            const l = parseInt(document.getElementById('w-log').value);

            const sum = (c + v + l) || 1;

            

            weights.cost = c / sum;

            weights.vibe = v / sum;

            weights.log = l / sum;


            document.getElementById('w-cost-label').innerText = Math.round(weights.cost * 100) + '%';

            document.getElementById('w-vibe-label').innerText = Math.round(weights.vibe * 100) + '%';

            document.getElementById('w-log-label').innerText = Math.round(weights.log * 100) + '%';

            render();

        }


        function addVenue() {

            venues.push({ id: Date.now(), name: "New Venue", capacity: 150, rental: 5000, catering: 100, vibe: 5, log: 5 });

            render();

        }


        function deleteVenue(id) {

            venues = venues.filter(v => v.id !== id);

            render();

        }


        function updateVenue(id, field, value) {

            const v = venues.find(v => v.id === id);

            if (field === 'name') v[field] = value;

            else v[field] = parseFloat(value) || 0;

            render();

        }


        function render() {

            const tbody = document.getElementById('table-body');

            const guests = parseInt(document.getElementById('globalGuests').value) || 0;

            const budget = parseFloat(document.getElementById('globalBudget').value) || 0;


            // Sort by match score

            const processed = venues.map(v => {

                const totalCost = v.rental + (v.catering * guests);

                

                // Score 0-10 based on how well it fits budget

                let costScore = 10 * (1 - (totalCost / (budget * 1.5)));

                costScore = Math.max(0, Math.min(10, costScore));


                const matchScore = (

                    (costScore * weights.cost) +

                    (v.vibe * weights.vibe) +

                    (v.log * weights.log)

                );


                return { ...v, totalCost, matchScore, overCapacity: guests > v.capacity, overBudget: totalCost > budget };

            });


            processed.sort((a, b) => b.matchScore - a.matchScore);


            tbody.innerHTML = processed.map(v => `

                <tr class="hover:bg-slate-50/50 transition">

                    <td class="p-4">

                        <input type="text" value="${v.name}" class="w-full font-bold text-slate-800 bg-transparent border-none p-0 focus:ring-0" onchange="updateVenue(${v.id}, 'name', this.value)">

                        <div class="flex items-center gap-2 mt-1">

                            <span class="text-[9px] text-slate-400 font-bold">MAX CAP:</span>

                            <input type="number" value="${v.capacity}" class="w-10 text-[10px] bg-slate-100 rounded border-none p-0.5" onchange="updateVenue(${v.id}, 'capacity', this.value)">

                            ${v.overCapacity ? '<span class="text-[9px] text-red-500 font-bold uppercase">Too Small</span>' : ''}

                        </div>

                    </td>

                    <td class="p-4">

                        <div class="flex items-center text-sm font-medium">

                            <span class="text-slate-300 mr-1">$</span>

                            <input type="number" value="${v.rental}" class="w-20 bg-transparent border-none p-0 focus:ring-0" onchange="updateVenue(${v.id}, 'rental', this.value)">

                        </div>

                    </td>

                    <td class="p-4">

                        <div class="flex items-center text-sm font-medium">

                            <span class="text-slate-300 mr-1">$</span>

                            <input type="number" value="${v.catering}" class="w-16 bg-transparent border-none p-0 focus:ring-0" onchange="updateVenue(${v.id}, 'catering', this.value)">

                        </div>

                    </td>

                    <td class="p-4">

                        <div class="text-sm font-bold ${v.overBudget ? 'text-red-500' : 'text-slate-700'}">

                            $${v.totalCost.toLocaleString()}

                        </div>

                        <div class="text-[9px] text-slate-400 font-medium">$${(v.totalCost/guests).toFixed(2)} p/p</div>

                    </td>

                    <td class="p-4">

                        <div class="flex flex-col gap-1 items-center">

                            <div class="flex items-center gap-1">

                                <span class="text-[8px] font-bold text-pink-400">VIBE</span>

                                <input type="number" value="${v.vibe}" class="w-8 text-[10px] font-bold bg-pink-50 text-pink-600 rounded text-center border-none p-0.5" onchange="updateVenue(${v.id}, 'vibe', this.value)">

                            </div>

                            <div class="flex items-center gap-1">

                                <span class="text-[8px] font-bold text-amber-400">LOGS</span>

                                <input type="number" value="${v.log}" class="w-8 text-[10px] font-bold bg-amber-50 text-amber-600 rounded text-center border-none p-0.5" onchange="updateVenue(${v.id}, 'log', this.value)">

                            </div>

                        </div>

                    </td>

                    <td class="p-4 bg-indigo-50/30 text-center">

                        <div class="inline-block px-3 py-1 rounded-full bg-indigo-600 text-white text-xs font-black">

                            ${v.matchScore.toFixed(1)}

                        </div>

                    </td>

                    <td class="p-4">

                        <button onclick="deleteVenue(${v.id})" class="text-slate-300 hover:text-red-400 transition">

                            <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" /></svg>

                        </button>

                    </td>

                </tr>

            `).join('');

        }


        function downloadCSV() {

            const guests = parseInt(document.getElementById('globalGuests').value);

            const budget = parseInt(document.getElementById('globalBudget').value);

            

            let csv = "Venue Name,Capacity,Rental Fee,Catering p/p,Total Est. Cost,Vibe Score,Logistics Score,Match Score\n";

            

            venues.forEach(v => {

                const totalCost = v.rental + (v.catering * guests);

                let costScore = 10 * (1 - (totalCost / (budget * 1.5)));

                costScore = Math.max(0, Math.min(10, costScore));

                const matchScore = ((costScore * weights.cost) + (v.vibe * weights.vibe) + (v.log * weights.log)).toFixed(2);

                

                csv += `"${v.name}",${v.capacity},${v.rental},${v.catering},${totalCost},${v.vibe},${v.log},${matchScore}\n`;

            });


            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            const url = URL.createObjectURL(blob);

            link.setAttribute("href", url);

            link.setAttribute("download", "Wedding_Venue_Comparison.csv");

            link.style.visibility = 'hidden';

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        }


        render();

    </script>
